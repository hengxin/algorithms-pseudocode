\begin{algorithm}[H]
  \caption{Selection Sort.}
  \label{alg:selection-sort}
  \begin{algorithmic}[1]
    \Procedure{Selection-Sort}{$A, n$}
      \For{$i \gets 1 \;\textbf{to}\; n-1$}
	\For{$j \gets i+1 \;\textbf{to}\; n$}
	  \If{$A[j] < A[i]$}
	    \State \Call{Swap}{$A[j], A[i]$}
	  \EndIf
	\EndFor
      \EndFor
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
