\begin{algorithm}[H]
  \caption{Bubblesort (Bubble the largest element each time.).}
  \label{alg:bubblesort-largest}
  \begin{algorithmic}[1]
    \Procedure{Bubblesort}{$A[1 \cdots n]$}
      \For{$i \gets n \;\text{\bf downto } 2$}
	\For{$j \gets 1 \;\text{\bf to } i-1$}
	  \If{$A[j] > A[j+1]$}
	    \State $\Call{Swap}{A[j], A[j+1]}$
	  \EndIf
	\EndFor
      \EndFor
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
